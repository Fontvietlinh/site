    <script>
    const blogId = "5243210247122196079";
    let startIndex = 1;
    const maxResults = 24;
    let currentLabel = '';
    let currentSearch = '';
    function setTheme(theme) {
    document.documentElement.setAttribute('data-theme', theme);
    localStorage.setItem('font-linh-theme', theme);
    }
    setTheme(localStorage.getItem('font-linh-theme') || 'paper');
    function fetchPosts() {
    const script = document.createElement('script');
    let url = `https://www.blogger.com/feeds/${blogId}/posts/default?alt=json-in-script&callback=render&start-index=${startIndex}&max-results=${maxResults}`;
    if (currentLabel) url = `https://www.blogger.com/feeds/${blogId}/posts/default/-/${currentLabel}?alt=json-in-script&callback=render&start-index=${startIndex}&max-results=${maxResults}`;
    if (currentSearch) url += `&q=${encodeURIComponent(currentSearch)}`;
    script.src = url; document.body.appendChild(script);
    }
    function render(data) {
    const container = document.getElementById('font-container');
    if (!data.feed.entry) { document.getElementById('load-more-btn').innerText = "Hết dữ liệu"; return; }
    data.feed.entry.forEach(post => {
    const title = post.title.$t;
    const blogLink = post.link.find(l => l.rel === 'alternate').href;
    const content = post.content.$t;
    const labels = post.category ? post.category.map(c => c.term) : [];
    let priceText = "";
    let isPaid = false;
    const priceMatch = content.match(/(Giá|Gia|Phí Việt hóa|Phí Việt Hóa):\s*([0-9.kđĐ]+)/i);
    if (priceMatch) {
    isPaid = true;
    // Đổi thành Phí Việt Hoá (Viết hoa chữ H) và giữ đ thường
    priceText = "Phí Việt Hoá: <span>" + priceMatch[2].toLowerCase() + "</span>";
    }
    let driveLink = "";
    const driveMatch = content.match(/https:\/\/drive\.google\.com\/[^\s"']+/);
    if (driveMatch) driveLink = driveMatch[0].replace(/&amp;/g, '&');
    const isVip = labels.some(l => l === 'FontTraPhi' || l === 'FontVip' || l.toUpperCase().includes('VIP')) || isPaid;
    let finalLink = isVip ? `https://m.me/FontVietLinh?text=Tôi muốn mua font: ${title}` : (driveLink ? driveLink : blogLink);
    const btnTxt = isVip ? "Mua ngay" : "Tải ngay";
    const btnCls = isVip ? "btn-paid" : "btn-free";
    let img = "https://via.placeholder.com/400x250?text=Font+Viet+Linh";
    if (content.includes('<img')) {
    const m = content.match(/src="([^"]+)"/);
    if (m) img = m[1];
    }
    container.innerHTML += `
    <div class="font-card">
    ${isVip ? '<div class="badge-vip">PREMIUM</div>' : ''}
    <img src="${img}" class="font-img" alt="${title}" loading="lazy">
    <div class="font-info">
    <div class="font-title">${title}</div>
    <div class="font-price">${priceText ? priceText : 'Miễn phí'}</div>
    <a href="${finalLink}" target="_blank" class="btn-download ${btnCls}">${btnTxt}</a>
    </div>
    </div>`;
    });
    startIndex += maxResults;
    }
    function loadByLabel(label) {
    currentLabel = label; currentSearch = '';
    document.querySelectorAll('nav a').forEach(a => a.classList.remove('active'));
    if (event && event.target.tagName === 'A') event.target.classList.add('active');
    resetAndFetch();
    }
    function handleSearch(e) { if (e.key === 'Enter') { currentSearch = document.getElementById('searchInput').value; currentLabel = ''; resetAndFetch(); } }
    function resetAndFetch() { document.getElementById('font-container').innerHTML = ''; startIndex = 1; document.getElementById('load-more-btn').innerText = "XEM THÊM FONT"; fetchPosts(); }
    fetchPosts();
    </script>
